«ከኢትዮጵያ ጋር እታረቃለሁ፤ አለበለዚያ። . .&#87፤

ሁሴን አይዲድ

ከሶማሊያ አንጃ መሪዎች ከፍተኛ ወታደራዊ ኃይል እንዳላቸው የሚነገርላቸው ሁሴን አይዲድ ከኢትዮጵያ ጋር ለመደራደርና ለመታረቅ ሲያስታውቁ፣ የኢትዮጵያ ሰራዊት ደግሞ ሞቃዲሾ አቅራቢያ ወደሚገኙት የአይዲድ የመጨረሻ ወታደራዊ ቤዞች እየተጠጋ መሆኑ ተሰማ።

የኤርትራ መንግስት በአይዲድ አማካይነት በሶማሊያ በኩል ሁለተኛውን ግንባር ከከፈተበት ከሚያዝያ 1991 ዓ.ም ጀምሮ የኢትዮጵያ ጦር የተጠናከረ ወታደራዊ እርምጃ በመውሰድ ላይ እንደሚገኝ ዲፕሎማቶች የአይን ምስክሮችና ሚዲያዎች ይገልፃሉ። ይሁንና አሁንም የኢትዮጵያ መንግስት አስተያየት ከመስጠት ተቆጥቧል። በተለይ ባለፉት ስምንት ወራት የኤርትራ መንግሥት ያሰለጠናቸውን የኦነግ ወታደሮች በአይዲድ አማካይነት ወደ ሶማሊያ ገብተው ባይዶዋ ላይ ከአይዲድ ሰራዊት ጋር ከሰፈሩ በኋላ እንዲሁም የአልኢትሃድ ታጣቂዎች በዚሁ ግንባር ተከማችተው ከኤርትራ የሚጋዘው መሳሪያም በነዚሁ ቦታዎች ተሰብስቦ ሁለቱ ድርጅቶች ከመሃል እስከ ምስራቅ ኢትዮጵያ ያለመረጋጋትን ለመፍጠር እንቅስቃሴ ለማድረግ በዝግጅት ላይ እያሉ የኢትዮጵያ ሰራዊት ግንቦት ወር 1991 ዓ.ም መጨረሻ ወደ ሶማሊያ 250 ኪ.ሜ በመግባት የኦነግንም ሆነ የአልኢትሃድን ታጣቂዎችና የተከማቸውን መሳሪያ መደምሰሱ ወይም መማረኩ ይታወሳል።

በሰኔ ወር 1991 ዓ.ም ደግሞ በዚሁ ጋዜጣ ላይ በተከታታይ እንደተዘገበው የኤርትራ መንግስት አሁንም የጦር መሳሪያ ወደ ሶማሊያ በማጋዙና ፀረ ኢትዮጵያ እንቅስቃሴ በመደረጉ የኢትዮጵያ ወታደሮች ወደፊት በመንቀሳቀስ የአይዲድ ጠንካራ ይዞታ የነበሩትን ጋርባሃሬንና ቡርዶቦን መያዛቸው ይታወሳል። ይህ በእንዲህ እንዳለ ነበር ባሳለፍነው ሳምንት ውስጥ የኢትዮጵያ ወታደሮች ፀረ- ኢትዮጵያ እንቅስቃሴን ለመግታት ወደ ሞቃዲሾ ሲጠጉ ሁሴን አይዲድ «ከኢትዮጵያ ጋር መታረቅ እፈልጋለሁ» በማለት ለዓለም ሕዝብ ያሳወቁት።

ስለኢትዮጵያ ሰራዊት እንቅስቃሴ የሶማሊያ ሚዲያዎችን፣ ዲፕሎማቶችንና የአይን ምስክሮችን ከሞቃዲሾ ጠቅሶ ዘገባውን ያሰራጨውን አዣንስ ፍራንስ ፕሬስ እንዳመለከተው «የኢትዮጵያ ወታደሮች በሰሜን ሶማሊያ፣ በሁሴን አይዲድ ይዞታ ስር የሚገኘውን ወታደራዊ ቦታዎች ለማጥቃት በዝግጅት ላይ ናቸው» ሲል ጠቅሷል። እንደዘገባው ከሆነ ጥቃቱ ያነጣጠረው ከሞቃዲሾ በስተምዕራብ 90 ኪ.ሜ ላይ የሚገኘውን የባሊዶሌ አየር ማረፊያና እዚያው አቅራቢያ የሚገኘውን የአይዲድ ሌጐ ወታደራዊ ቤዝን ነው። ከዚሁ ጋር የኪስማዩ ወደብም የጥቃቱ አንደኛው እቅድ መሆኑ ተሰምቷል።

እንደአካባቢው ወታደራዊ ተንታኞች ባሊዶሌና ሌጎ፣ ለአይዲድ የመጨረሻ ምሽጋቸው ከሆነው ከሞቃዲሾ ውጪ ያላቸውን የመጨረሻዎቹ ጠንካራ ይዞታዎች ባሊዶሌና ሌጎ ናቸው። በተለይ ባሊዶሌ ለአይዲድ ወሳኝ የትጥቅና የስንቅ መግቢያና ማከማቻ ናቸው። እንደ ሚዲያ ሰዎች ዘገባ ከኤርትራ የሚገቡ የጦር መሳሪያዎች ከሚከማቹበት ቦታዎች ዋናዎቹ እነዚህ ናቸው።

ከሞቃዲሾ የሚወጡት ዘገባዎች እንዳመለከቱት የኢትዮጵያ ጦር አንደኛው ክፍል በስተሰሜን ምስራቅ ሐምሌ 6/91 ዶሎው በተባለ ቦታ በኩል እንደተንቀሳቀሰ ሁለተኛው በስተደቡብ ምስራቅ፣ የጌዶ ክልልን አልፎ ባይ ክልል በመግባት በባኮል አካባቢ እንደተንቀሳቀሰ ገልፀዋል። የሁለቱም ርቀት ከሞቃዲሾ 100 ኪሎ ሜትር ዙሪያ እንደሆነ የተያያዘው ዜና አመልክቷል።

በሌላ በኩል የአይዲድ ተዋጊዎችን የኦነግ ታጣቂዎች እንደተቀላቀሉ የአይን ምስክሮችንና ሚሊሺያዎችን ዋቢ አድርገው ሚዲያዎች ቢገልፁም አይዲድ አሁንም አስተባብለዋል። እንደ ዓይን ምስክሮች ገለፃ በባሊዶሌ አየር ማረፊያና በአካባቢው የሚገኙት ሶማሊያውያን አንፃራዊ ሰላም ወዳለባት ሞቃዲሾ ተሰደዋል።

በአሁኑ ወቅት የአይዲድ ቀንደኛ ተቃዋሚና «ባይዶዋ በቁጥጥሬ ስር ነች» የሚለው የርርኣ ሰራዊት «ከባይዶሌ አየር ማረፊያ አጠገብ ከሚገኘው ሌጐ የአይዲድ ጠንካራ ወታደራዊ ይዞታ አቅራቢ ልዩ ስሙ ዋንላዌይን መንደር እንደከተመ ታውቋል።

አይዲድ አጣብቂኝ ውስጥ እንደገቡ መረጃዎች ካመለከቱና የእሳቸውም ቃል አቀባይ ኢብራሂም ጃማ «የኢትዮጵያ ወታደሮች ወደ ሶማሊያ እምብርት ዘልቀው ገብተዋል» ካሉ በኋላ ነበር አይዲድ ከኢትዮጵያ ጋር እርቅ የጠየቁት። ባሳለፍነው ሳምንት ሁሴን አይዲድ ቃለ መጠየቅ ላደረገላቸው የሬውተር ዘጋቢ «ከኢትዮጵያ ጋር መደራደርና መታረቅ እፈልጋለሁ፤ የዓለም አቀፉ ማህበረሰብ ኢትዮጵያ ጦርዋን ከሶማሊያ እንድታስወጣ ጫና ያድርግ» ብለዋል። ይሁንና የሶማሌ ላንድና የፑትላንድ እንዲሁም ቀሪዎቹ የሶማሊያ አንጃዎች ስለኢትዮጵያ በቦታው መገኘትም ሆነ «አድርጋለች/ እያደረገች ነው» የሚባለውን እንቅስቃሴ አልተቃወሙም። እንደውጭ ዲፕሎማቶች «የኢትዮጵያ እርምጃ ራስን ከጥቃት ከማዳን በላይም በሶማሊያ ውስጥ ዘላቂ ሰላምን፣ አይዲድን በማንበርከክ የሚያስገኝ ነው» ይላሉ።

አይዲድ «ከኢትዮጵያ ጋር መታረቅ እፈልጋለሁ» ብለው በኦፌሴል ቢገልፁም ናይሮቢ ላይ ለሚዲያ ሰዎች ዛቻ የተቀላቀለበት አስተያየታቸውን ሰጥተዋል። በቃለምልልሳቸው «አለበለዚያ ኢትዮጵያ ከኛ ጋር በሰላም ካልኖረች ትጐዳለች፣ ኢትዮጵያ ውስጥ አምስት ሚሊዮን የሚጠጉ ንፁህ ሶማሊያውያን አሉ። ስለዚህ ጦርነቱ ከሶማሊያ ድንበር ውጪ ሊሆንም ይችላል» ብለዋል። አያይዘውም «በርግጠኝነት ለመናገር ከፍተኛ ኪሳራ የሚደርስባቸው ኢትዮጵያውያን ናቸው። ምክንያቱም በከፍተኛ ሁኔታ ሊነኳኰቱ የሚችሉ ናቸውና፤ ወደ ጦርነት ስንመጣ እኛ ሶማሊያውያን የጦር ባለሙያዎች ነን» ብለዋል። አይዲድ ይህን ቢሉም «ኤርትራን ግብፅንና ሊቢያን የኢትዮጵያን ኃይል ከይዞታቸው እንዲያስወጡላቸው እርዳታቸውን ተማፅነዋል። በዘመቻ ፀሐይ ግባት ወቅት የኤርትራ መንግስት ከባድመ መውጣት ላይ እያለ ለነዚሁና ለሌሎች የአረብ ሀገራት ኢትዮጵያን ከባድመ ለማስወጣት እርዳታ እንዲያደርጉለት የኤርትራ መንግሥት መማፀኑ ይታወሳል።

ይህ በእንዲህ እንዳለ የአሶሽየትድ ዘጋቢ ኦስማን ሀሰን ከሞቃዲሾ ባስተላለፈው ዘገባ ካሁን በፊት ኤርትራ የጦር መሣሪያና የኦነግ ወታደሮችን ካስገባችበት ወደብ አንዱ ወደ ሆነው ኮርዮሌይ ወደብ የኢትዮጵያ ሰራዊ እየተቃረበ መሆኑን የአይን ምስክሮችን ዋቢ አድርጎ ጠቅሷል። ኮርዮሌይ ከሞቃዲሾ በስተምዕራብ 67 ኪ.ሜ ርቀት ላይ የምትገኝ ወደብ ነች።

በተመሳሳይ ሁኔታም የኢትዮጵያ ሠራዊት በሳምንቱ መጨረሻ ላይ በደቡብ ሶማሊያ በታችኛው ሸበሌ ክልል በሚገኘው ፋርሶሌይ መንደር መግባቱንና ከነዋሪው ጋር መገናኘቱን የአዣንስ ፍራንስ ፕሬስ ዘገባ አመልክቷል። የአይዲድ ሠራዊት እስካሁን እንዳልተጠጋ የተጠቀሰው ዘገባ ፋርሶሌይ አይዲድ ከሙዝ ምርት ከፍተኛ ገቢ ከሚያገኙበት ከቆርዬሎ አውራጃ 20 ኪሎ ሜትር ርቀት ላይ እንደምትገኝ ታውቋል። በተያያዘ ዜና ሁሴን አይዲድ በሰሜን ሞቃዲሾ ለደጋፊዎቻቸው ባደረጉት ንግግር «ከኢትዮጵያ ጋር ያለንን ግጭት ለማስወገድ ሊቢያ ጥረት የጀመረች ሲሆን በቅርቡ መፍትሔ ይገኛል» ብለዋል። የቢቢሲ ዘገባ እንዳመለከተው ደግሞ አይዲድ ወደ ሊቢያ ለመሔድ እንዳቀዱ መግለፃቸውን አመልክቶ ለተፈጠረው ግጭት በሊቢያ አማካኝነት መፍትሔ ይገኛል የሚል ተስፋ እንዳላቸው ገልጿል። አይዲድ ለደጋፊዎቻቸው ባደረጉት ገለፃ «የኢትዮጵያ ወታደሮች ከሞቃዲሾ 180 ኪሎ ሜትር ላይ ይገኛሉ» ብለዋል። ይሁንና የርርኣ ጦረኞች ደግሞ ከ90 ኪሎ ሜትር ባነሰ ርቀት ላይ እንደሚገኙ ሌሎች ሚዲያዎች መግለፃቸው ይታወሳል።

ከሪፖርተር ሐምሌ 12 ቀን 1991
