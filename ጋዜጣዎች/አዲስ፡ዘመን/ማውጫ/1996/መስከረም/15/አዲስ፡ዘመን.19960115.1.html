በብሩንዲ የሚሠማራው የኢትዮጵያ የሰላም አስከባሪ ሻለቃ በክብር ተሸኘ

በቡሩንዲ ዘላቂ ሰላም ለማምጣት እየተካሄደ ያለውን የሰላም ጥረት ለማገዥ ወደ ሥፍራው የሚጓዘው የኢትዮጵያ የሰላም አስከባሪ ሻለቃ የተጣለበትን ኃላፊነት በብቃት እንደሚወጣ ያላቸውን እምነት ጠቅላይ ሚኒስትር መለስ ዜናዊ አስታወቁ።

ወደ ቡሩንዲ የሚሠማራው የኢትዮጵያ የሰላም አስከባሪ ሻለቃ ትናንት በምድር ኃይል ቅጥር ግቢ በክብር በተሸኘበት ወቅት የኢፌዲሪ ጠቅላይ ሚኒስትርና የጦር ኃይሎች ጠቅላይ አዛዥ አቶ መለስ ዜናዊ በሰጡት መምሪያ የሰላም ጥበቃ ሻለቃው ከዚህ ቀደም ካካበተው ልምድ፣ ካለው ጀግንነትና ሕዝባዊነት በመነሳት የተሰጠውን ተልዕኮ በአስተማማኝ ሁኔታ እንደሚወጣ እምነታቸው እንደሆነ ገልጸዋል።

እህት በሆነችው ቡሩንዲ ለረጅም ጊዜ የቆየውን የእርስ በርስ ጦርነት በሰላም ለመፍታት የአፍሪካ ኅብረት የተጠናከረ ዲፕሎማሲያዊ ጥረት ሲያደርግ መቆየቱን የጠቆሙት ጠቅላይ ሚኒስትሩ፣ ችግሩን ለማስወገድ የሰላም አስከባሪ ኃይል እንዲመደብ መወሰኑንና ለዚህም ኅብረቱ ለአባል ሀገሮች ጥሪ ማቅረቡን አስታውሰዋል።

ኢትዮጵያ የአፍሪካ ኅብረት ዋና ከተማ ከመሆኗ ጋር በተያያዘ ኅብረቱ በማናቸውም ወቅት ከአገራችን ጎን የሚሰለፍ ታላቅ ድርጅት በመሆኑ በሰላም ማስከበሩ ሂደት አንድ ሻለቃ ጦር ወደ ሥፍራው እንዲጓዝ መንግሥት በወሰነው መሠረት ተገቢው ዝግጅት ሲደረግ መቆየቱን ገልጸዋል።

በብሩንዲ ወንድምና እህት ሕዝቦች ያሉበት ቢሆንም፤ እጅግ የተወሳሰበ ፖለቲካዊ ችግርና በየጊዜው የሚያገረሽ ግጭት ያለበት ስለሆነ የጀግንነት መንፈሳችሁና ሕዝባዊነታችሁ ከመቼውም ጊዜ በላይ ተፈላጊ የሚሆንበት ቦታ ነው ብለዋል።

የሰላም አስከባሪው ኃይል በሀገሪቱ ያለውን ውስብስብ ፖለቲካዊና ወታደራዊ ችግር በተለመደው ሕዝባዊነት፤ ትዕግሥትና ጀግንነት ተቋቁሞ የተጣለበትን ኃላፊነት በብቃት እንደሚወጣ እምነታቸውን ደጋግመው ያረጋገጡት ጠቅላይ ሚኒስትር መለስ፤ ለሰራዊቱ መልካም ጉዞና የተቃና ሥራ እንዲያጋጥመው ከልብ ተመኝተዋል።

በሻለቃው ውስጥ ያለው ሠራዊት ከ800 እስከ አንድ ሺ እንደሚሆን የተናገሩት የጦሩ የዘመቻና ሥልጠና ኃላፊ ሻለቃ አሥራት በልሁ በበኩላቸው፤ ሠራዊቱ ኃላፊነቱን በብቃት ለመወጣት በአገር ውስጥና ከውጭ በመጡ ወታደራዊ ባለሙያዎች ሰፊ ሥልጠና እንደተሰጠው ገልጸዋል።

በክብር አሸኛኘት ፕሮግራሙ ላይ ሚኒስትሮች፣ አምባሳደሮች፣ የአገር መከላከያ ሠራዊት ከፍተኛ የጦር አዛዦች እንዲሁም ጥሪ የተደረገላቸው እንግዶች የተገኙ ሲሆን፣ የሰላም አስከባሪው ሻለቃ በክብር እንግዶቹ ፊት በወታደራዊ ማርሽ እየታጀበ የሰልፍ ትርዒት አሳይቷል።

ከአዲስ ዘመን መስከረም 15 ቀን 1996
